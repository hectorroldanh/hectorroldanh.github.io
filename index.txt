  <script>
    (function() {
      document.addEventListener('DOMContentLoaded', function() {
        // Carrusel
        const slides = document.querySelectorAll('.carrusel img');
        let current = 0;
        if (slides.length > 1) {
          setInterval(() => {
            slides[current].classList.remove('active');
            slides[current].setAttribute('aria-hidden', 'true');
            current = (current + 1) % slides.length;
            slides[current].classList.add('active');
            slides[current].setAttribute('aria-hidden', 'false');
          }, 5000);
        }

        // Menú móvil
        const toggle = document.querySelector('.menu-toggle');
        const menu = document.querySelector('nav.menu ul');
        toggle && toggle.addEventListener('click', function() {
          const expanded = this.getAttribute('aria-expanded') === 'true';
          this.setAttribute('aria-expanded', String(!expanded));
          menu.classList.toggle('show');
        });

      // Submenús en móvil: abrir tocando el item padre
      document.querySelectorAll('nav.menu li').forEach(li => {
        const submenu = li.querySelector('.submenu');
        if (submenu) {
          li.querySelector('a').addEventListener('click', function(e){
            if (window.innerWidth <= 768) {
              e.preventDefault();
              submenu.classList.toggle('show');
            }
          });
        }
      });

        // Smooth scroll
        document.querySelectorAll('a[href^="#"]').forEach(a => {
          a.addEventListener('click', function(e) {
            const href = this.getAttribute('href');
            if (!href || href === '#') return;
            e.preventDefault();
            const target = document.querySelector(href);
            if (target) {
              if (menu && menu.classList.contains('show')) {
                menu.classList.remove('show');
                toggle.setAttribute('aria-expanded', 'false');
              }
              const headerOffset = document.querySelector('header').offsetHeight || 72;
              const rect = target.getBoundingClientRect();
              const top = rect.top + window.pageYOffset - headerOffset + 6;
              window.scrollTo({ top, behavior: 'smooth' });
            }
          });
        });

        // Reveal animation
        const reveals = document.querySelectorAll('.reveal');
        if ('IntersectionObserver' in window) {
          const io = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
              if (entry.isIntersecting) {
                entry.target.classList.add('active');
                io.unobserve(entry.target);
              }
            });
          }, { threshold: 0.12 });
          reveals.forEach(r => io.observe(r));
        } else {
          function revealOnScroll() {
            reveals.forEach(el => {
              const top = el.getBoundingClientRect().top;
              if (top < window.innerHeight - 120) el.classList.add('active');
            });
          }
          window.addEventListener('scroll', revealOnScroll);
          revealOnScroll();
        }

        // Formulario
        const form = document.getElementById('form-contacto');
        if (form) {
          form.addEventListener('submit', function() {
            alert('¡Gracias por tu mensaje! Te contactaré pronto.');
          });
        }

        // Back to top
        const back = document.getElementById('back-to-top');
        window.addEventListener('scroll', () => {
          back.style.display = window.scrollY > 300 ? 'block' : 'none';
        });
        back.addEventListener('click', e => {
          e.preventDefault();
          window.scrollTo({ top: 0, behavior: 'smooth' });
        });

        // Modales
        let lastFocused = null;
        function abrirModal(id) {
          const modal = document.getElementById(id);
          if (!modal) return;
          lastFocused = document.activeElement;
          modal.setAttribute('aria-hidden', 'false');
          modal.style.display = 'flex';
          const btnClose = modal.querySelector('.cerrar');
          if (btnClose) btnClose.focus();
          document.body.style.overflow = 'hidden';
        }
        function cerrarModal(id) {
          const modal = document.getElementById(id);
          if (!modal) return;
          modal.setAttribute('aria-hidden', 'true');
          modal.style.display = 'none';
          if (lastFocused) lastFocused.focus();
          document.body.style.overflow = '';
        }
        window.abrirModal = abrirModal;
        window.cerrarModal = cerrarModal;

        // Cerrar modales al hacer clic fuera o en botón cerrar
        document.querySelectorAll('.cerrar').forEach(btn => {
          btn.addEventListener('click', () => {
            const id = btn.dataset.modal;
            if (id) cerrarModal(id);
          });
        });
        window.addEventListener('click', e => {
          document.querySelectorAll('.modal').forEach(mod => {
            if (e.target === mod) {
              mod.setAttribute('aria-hidden', 'true');
              mod.style.display = 'none';
              document.body.style.overflow = '';
            }
          });
        });

        // Cerrar modales con ESC
        document.addEventListener('keydown', e => {
          if (e.key === 'Escape') {
            document.querySelectorAll('.modal[aria-hidden="false"]').forEach(modal => {
              modal.setAttribute('aria-hidden', 'true');
              modal.style.display = 'none';
              document.body.style.overflow = '';
            });
          }
        });

        // Exponer abrir/cerrar para compatibilidad (algunos botones usan onclick inline)
        window.abrirModal = abrirModal;
        window.cerrarModal = cerrarModal;

        /* -------------------
           Evita errores si se ejecuta antes de DOM
        ------------------- */
      }); // Fin de DOMContentLoaded
    })();
  </script>
