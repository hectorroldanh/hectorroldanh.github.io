        // Variables para controlar la paginación
        let obraActual = '';
        let paginaActual = 0;

        // Funciones para los modales
        function abrirModalLectura(obraId) {
            obraActual = obraId;
            paginaActual = 0;
            
            const modal = document.getElementById('modal-lectura');
            const contenido = document.getElementById('contenido-lectura');
            const obra = contenidoObras[obraId];
            
            if (obra) {
                // Mostrar la primera página
                contenido.innerHTML = obra.paginas[0];
                
                // Actualizar indicador de página
                document.getElementById('pagina-indicador').textContent = `1 / ${obra.paginas.length}`;

                
                // Configurar enlace de compra
                document.getElementById('enlace-compra').href = obra.enlaceCompra;
                
                // Habilitar/deshabilitar botones de navegación
                document.getElementById('btn-anterior').disabled = true;
                document.getElementById('btn-siguiente').disabled = obra.paginas.length <= 1;
            } else {
                contenido.innerHTML = '<p>Contenido no disponible</p>';
            }
            
            modal.style.display = 'block';
            document.body.style.overflow = 'hidden';
            
            // Mostrar controles de lectura
            document.getElementById('controles-lectura').style.display = 'flex';
        }

        function cerrarModalLectura() {
            const modal = document.getElementById('modal-lectura');
            modal.style.display = 'none';
            document.body.style.overflow = 'auto';
            
            // Ocultar controles de lectura
            document.getElementById('controles-lectura').style.display = 'none';
        }

        // Función para cambiar de página
        function cambiarPagina(direccion) {
            if (!obraActual) return;
            
            const obra = contenidoObras[obraActual];
            paginaActual += direccion;
            
            if (paginaActual < 0) paginaActual = 0;
            if (paginaActual >= obra.paginas.length) paginaActual = obra.paginas.length - 1;
            
            const contenido = document.getElementById('contenido-lectura');
            contenido.innerHTML = obra.paginas[paginaActual];
            
            // Actualizar indicador de página
            document.getElementById('pagina-indicador').textContent = `${paginaActual + 1} / ${obra.paginas.length}`;
            
            // Habilitar/deshabilitar botones de navegación
            document.getElementById('btn-anterior').disabled = paginaActual === 0;
            document.getElementById('btn-siguiente').disabled = paginaActual === obra.paginas.length - 1;
            
            // Restaurar el tamaño de fuente si se había modificado
            const tamanoActual = parseInt(window.getComputedStyle(contenido).fontSize);
            if (tamanoActual !== 19) {
                contenido.style.fontSize = '19px';
            }

    // 👈 Fuerza el scroll al inicio cada vez que se cambia de página
    contenido.scrollTop = 0;
    
    // opcional: si el scroll está en el modal entero, también:
    const modal = document.getElementById('modal-lectura');
    modal.scrollTop = 0;

        }

        // Controles de lectura
        function cambiarTamano(accion) {
            const contenido = document.getElementById('contenido-lectura');
            const tamanoActual = parseInt(window.getComputedStyle(contenido).fontSize);

            
            if (accion === 'aumentar' && tamanoActual < 24) {
                contenido.style.fontSize = (tamanoActual + 2) + 'px';
            } else if (accion === 'disminuir' && tamanoActual > 14) {
                contenido.style.fontSize = (tamanoActual - 2) + 'px';
            }
        }

        function mostrarMensaje(mensaje) {
            // Crear un elemento de notificación
            const notificacion = document.createElement('div');
            notificacion.style.cssText = `
                position: fixed;
                top: 20px;
                left: 50%;
                transform: translateX(-50%);
                background: var(--color-accent);
                color: var(--color-dark);
                padding: 15px 25px;
                border-radius: 30px;
                font-weight: 600;
                z-index: 10001;
                box-shadow: 0 4px 15px rgba(0,0,0,0.2);
            `;
            notificacion.textContent = mensaje;
            
            document.body.appendChild(notificacion);
            
            // Eliminar después de 3 segundos
            setTimeout(() => {
                notificacion.style.opacity = '0';
                notificacion.style.transition = 'opacity 0.5s';
                setTimeout(() => {
                    document.body.removeChild(notificacion);
                }, 500);
            }, 3000);
        }
